library(readxl)
fall <- readxl::read_xlsx("./SCEQ received Viz LMS.xlsx",sheet=2)

#adding faculty column
fac <- read.csv("faculty and programs.csv",header=T)

  fall %>%
    left_join(fac, by = "Programs") -> fall #adding faculty and Department col
